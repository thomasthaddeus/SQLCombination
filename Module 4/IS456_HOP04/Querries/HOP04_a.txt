-- 03 SUBSTR
-----------------------------------
SELECT SUBSTR('this string', 6);
SELECT SUBSTR('this string', 6, 3);
SELECT released,
SUBSTR(released, 1, 4) AS year,
SUBSTR(released, 6, 2) AS month,
SUBSTR(released, 9, 2) AS day
FROM album
ORDER BY released;
